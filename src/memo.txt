function associative_array_definition($input) {
    //$colorに連想配列（3つ）のデータを代入。(今回想定している値で使用する配列は'melon' => 'green')
    $color = [    # 変わらない定義だから定数にするべき
        'apple' => 'red',
        'banana' => 'yellow',
        'melon' => 'green',
    ];
    //変数に想定しているkey(melon)を代入
    $input = $color['melon']; #  input使おう  $color['melon'] で"green" を受け取って何がしたいか謎
    //引数で指定した値と一致したら戻り値でgreenを出力
    //forで$iをの初期値0を用意。。$iが＄colorに格納されている配列の数以下の場合は＄iにインクリメント
    for ($i =0; $i < count($color); $i++){ # 連想配列? mapとかdictionaryのkey valueだけど、phpでなんて呼ぶかは知らない. ただkeyをスマートに回す方法あるはず
        //もし想定している$colorのkey(melon)とforで回している$inputのkeyが一致してればtrue
        if($color['melon'] == $input[$i]){  # もはや謎 inputの中身がmelonはfunctionの予備出し時に確定しいてそのinputとkeyが同じ時に中身のgreenを呼び出してreturnしよう
            //returnで想定しているデータを$inputに返す
            echo $input[$i];  # functionの外でechoしよう
            return $input[$i];  # inputから取り出すのが謎 colorから取り出そう
        }
    }
    //string(5) "green"が出力
    var_dump($input);  # これは何用???keyがマッチしなかった時用?
}
    //関数を実行
    associative_array_definition(['melon']); # なんでarray?
